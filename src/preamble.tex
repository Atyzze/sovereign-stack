% ===================== THE SOVEREIGN STACK, VOL 1 - preamble =====================
% Compile with xelatex (fonts set via fontspec by the pandoc template).

% ---- Page geometry: A4 by default, A5 variant selected by src/paper.tex ----
% build.sh writes src/paper.tex (the same generated-file pattern as
% src/version.tex): it picks the paper class, issues the matching \geometry call,
% sets the version-tag corner coordinates (\sstagx, \sstagy), and sets the
% figure-fit flag \sspaperclass. With no paper.tex present (a direct latexmk on
% the generated .tex), the A4 fallback below runs, so the .tex still builds
% standalone exactly as the A4 book does. Same manuscript, same machinery: only
% the geometry differs between the A4 edition and its A5 variant.
\usepackage{geometry}
\IfFileExists{src/paper.tex}{\input{src/paper.tex}}{%
  \def\sspaperclass{a4}%
  \geometry{a4paper,top=2.6cm,bottom=2.5cm,left=2.7cm,right=2.7cm,headheight=15pt}%
  \def\sstagx{20cm}\def\sstagy{28.7cm}}

% ---- Unicode glyph fallbacks (defensive; manuscript is pure ASCII) ----

\newunicodechar{→}{\ensuremath{\rightarrow}}

% ---- version tag, auto-derived from the folder name by build.sh ----
% build.sh writes src/version.tex (a single line: \def\ssversion{v1v1i<n>})
% from the name of the folder the project sits in, so this tag updates itself
% on every build with no manual edit. The placeholder below is only used if the
% .tex is compiled directly (without build.sh); i00 flags "version unknown".
\def\ssversion{v1v1i00}
\IfFileExists{src/version.tex}{\input{src/version.tex}}{}
\AddToShipoutPictureFG{\put(\LenToUnit{\sstagx},\LenToUnit{\sstagy}){\makebox(0,0)[r]{\ttfamily\footnotesize\color{softblack!50}\ssversion}}}

% ---- Figure-fit helper for the A5 variant ----
% Wraps a wide box (the two TikZ roadmap diagrams). On the A4 edition this is a
% pure no-op, so A4 output is byte-for-byte unchanged. On A5 it scales the box
% down to the text width only when it would otherwise overflow (never enlarges),
% keeping the diagrams inside the narrower A5 margins.
\newcommand{\ssfit}[1]{%
  \ifdefstring{\sspaperclass}{a5}%
    {\resizebox{\ifdim\width>\linewidth\linewidth\else\width\fi}{!}{#1}}%
    {#1}}

% ---- Pointer boxes (markdown blockquotes -> accent-barred callouts) ----

    boxrule=0pt, leftrule=2.6pt, arc=0pt, outer arc=0pt,
    left=11pt, right=11pt, top=8pt, bottom=8pt,
    before skip=11pt, after skip=11pt]\small\color{softblack}}{%
  \end{tcolorbox}}
